class cliente{
    private:
        int id;
        string nom;
        string DNI;
        string correo;
        string direccion;
    public:
        cliente(int _id = 0, string _nom = "", string _DNI = "", string _correo = "", string _direccion = "")
            : id(_id), nom(_nom), DNI(_DNI), correo(_correo), direccion(_direccion){}
        int getid() const {
            return id;
        }
        string getnom() const {
            return nom;
        }
        string getDNI() const {
            return DNI;
        }
        string getcorreo() const {
            return correo;
        }
        string getdireccion() const {
            return direccion;
        }
        void setnom(const string& nuevoNombre) {
            nom = nuevoNombre;
        }
        void setDNI(const string& nuevoDNI) {
            DNI = nuevoDNI;
        }
        void setcorreo(const string& nuevoCorreo) {
            correo = nuevoCorreo;
        }
        void setDireccion(const string& nuevaDireccion) {
            direccion = nuevaDireccion;
        }
        static cliente cargar(const string& linea) {
            vector<string> partes;
            size_t  inicio = 0, fin;
            while ((fin = linea.find(',', inicio)) != string::npos) {
                partes.push_back(linea.substr(inicio, fin - inicio));
                inicio = fin++;
            }
            partes.push_back(linea.substr(inicio));
            return cliente(
                stoi(partes[0]), partes[1], partes[2], partes[3], partes[4]
            );
        }
};

void guardarCSV(const vector<cliente>& lista_cliente, const vector<ventas>& lista_ventas, const vector<gastos_realizados>& lista_gastos, const vector<pagos_realizados>& lista_pagos, const vector<salarios>& lista_salarios, const string& nombreArchivo) {
    ofstream archivo(nombreArchivo);

    archivo << "== CLIENTES ==\n";
    archivo << "ID, Nombre, DNI, Correo, DirecciÃ³n\n";
    for (const auto& est : lista_cliente) {
        archivo << est.getid() << "," << est.getnom() << "," << est.getDNI() << "," << est.getcorreo() << "," << est.getdireccion() << "\n";
    }
    archivo << "== VENTAS ==\n";
    archivo << "ID, Nombre del Producto, Precio del Producto, Descuento, Metodo de pago, Fecha de venta, Estado de venta\n";
    for (const auto& est : lista_ventas) {
        archivo << est.getid() << "," << est.getnomprod() << "," << est.getprecio() << "," << est.getdesc() << "," << est.getmetpago() << "," << est.getfechaventa() << "," << est.getestadoventa() << "\n";
    }
    archivo << "== GASTOS ==\n";
    archivo << "ID, Detalle del gasto, Fecha de gasto, Estado del gasto\n";
    for (const auto& est : lista_gastos) {
        archivo << est.getid() << "," << est.getgastodetalle() << "," << est.getfechagasto() << "," << est.getestadogasto() << "\n";
    }
    archivo << "== PAGOS ==\n";
    archivo << "ID, Detalle del pago, Tipo de pago, Monto del pago, Fecha de pago, Estado del pago\n";
    for (const auto& est : lista_pagos) {
        archivo << est.getid() << "," << est.getpagodetalle() << "," << est.gettipopago() << "," << est.getmonto() << "," << est.getfechapago() << "," << est.getestadopago() << "\n";
    }
    archivo << "== SALARIOS ==\n";
    archivo << "ID, Monto del salario, Bono, Nombre del empleado, Detalle del salario, Estado del salario\n";
    for (const auto& est : lista_salarios) {
        archivo << est.getid() << "," << est.getmontosalario() << "," << est.getbono() << "," << est.getnomempleado() << "," << est.getdetallesalario() << "," << est.getestadosalario() << "\n";
    }
    archivo.close();
}

int main()
{
    vector <cliente> lista_cliente = { cliente(1, "Luis", "12345678", "luis@mail.com", "Av Lima") };
    vector <ventas> lista_ventas = { ventas(1, "Laptop", 2500, 10, "Tarjeta", "2025-01-11", "Pagado") };
    vector <gastos_realizados> lista_gastos = { gastos_realizados(1, "Luz", "2025-01-01", "Realizado") };
    vector <pagos_realizados> lista_pagos = { pagos_realizados(1, "Proveedor", "Transferencia", 1500, "2025-01-05", "Realizado") };
    vector <salarios> lista_salarios = { salarios(1, 3000, 200, "Carlos", "Sueldo Enero", "Pagado") };
    guardarCSV(lista_cliente, lista_ventas, lista_gastos, lista_pagos, lista_salarios, "Reporte.CSV");
    cout << "Archivo Excel Generado correctamente \n";
    return 0;
}
